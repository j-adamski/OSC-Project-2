import t3

    def test_t3Complete(self):
        self.assertRaises(FileNotFoundError, t3.t3Complete, "homework3")

    def test_t3Report(self):
        self.assertRaises(FileNotFoundError, t3.t3Complete, "homework3")

    def test_t3Add(self):
        self.assertRaises(FileNotFoundError, t3.t3Complete, "test project")

    def test_t3Delete(self):
        self.assertRaises(FileNotFoundError, t3.t3Complete, "test project")

    def test_t3Edit(self):
        self.assertRaises(FileNotFoundError, t3.t3Complete, "homework3")

    def test_t3Start(self):
        self.assertRaises(FileNotFoundError, t3.t3Complete, "homework3")

    def test_t3Stop(self):
        self.assertRaises(FileNotFoundError, t3.t3Complete, "homework3")

    if __name__ == '__main__':
        unittest.main()
